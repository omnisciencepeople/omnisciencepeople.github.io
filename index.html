求1+1/2+...+1/100

#include<stdio.h>
int main()
{
int i;
float sum=0;
for(i=1;i<=100;i++)
sum=sum+1.0/i;//或者为sum=sum+1/float（i）；
printf(“sum=%f\n”,sum);
return 0;
}
/*
整形数据/整形数据，结果为整形；
除法运算符中若有至少一个浮点型，结果为浮点型；
商的符号由除数和被除数决定
因此对上述sum求值过程中进行强制类型转换；

取余运算符符号只和被除数符号相同；
*/

如何看懂程序

分三步走：
第一步：流程控制，也就是程序的执行顺序；
第二步：语句功能，即这个语句种类及其作用；
第三步：试数，这个是看懂程序的关键；

在我们学习过程中可能会遇到一些小算法，我们可以尝试去对它编程，在解决不了的时候，可能会陷入自我怀疑中，这不是正确的学习方式；
这时候我们应该去看答案，重要的是把答案看懂，之后照着答案去敲程序并进行调试；如果可以，去修改程序，看看输出结果有什么不同；
如果连答案都看不懂，就把它背下来，不断去想，不断去敲，这是一个自我蜕变的过程，惟其坚持，方可成功！

对三个数字从小到大排序

#include<stdio.h>
int main()
{
int a,b,c;
printf(“请对这三个数赋值(中间以空格隔开):”);
scanf(“%d %d %d”,&a,&b,&c);
int t;
if(a>b)
{
t=a;
a=b;
b=t;
}
if(a>c)
{
t=a;
a=c;
c=t;
}
if(b>c)
{
t=b;
b=c;
c=t;
}
printf(“输出排序后的结果:%d %d %d\n”,a,b,c);
return 0;
}

if求分数等级

#include<stdio.h>
int main()
{
float score;
char ch;
do
{
printf(“请输入你的成绩:”);
scanf(“%f”,&score);
if(score>100)
printf(“做梦!\n”);
else if(80<=score&&score<=100)
printf(“优秀!\n”);
else if(70<=score&&score<80)
printf(“良好!\n”);
else if(60<=score&&score<70)
printf(“及格!\n”);
else if(0<=score&&score<60)
printf(“挂科!\n”);
else
printf(“你输入的成绩错误!\n”);
printf(“你是否还要继续（Y/N）:\n”);
scanf(“ %c”,&ch);//这里%c前加一个空格才可实现人机交互；
}while(ch==’Y’);
return 0;
}

程序执行

#include<stdio.h>
int main()
{
int m,k=20;
m=(3>2)&&(k=8);
printf(“m=%d,k=%d\n”,m,k);
m=(1>2)&&(k=5);
printf(“m=%d,k=%d\n”,m,k);
m=(1>2)||(k=3);
printf(“m=%d,k=%d\n”,m,k);
m=(2>1)||(k=1);
printf(“m=%d,k=%d\n”,m,k);
m=(2>1)&&(k=0);
printf(“m=%d,k=%d\n”,m,k);
return 0;

C语言printf用法

#include <stdio.h>
#include <stdlib.h>

int main()
{
printf(“Hello world!\n”);
int i=5,j=1;
printf(“%d\n”,i);
printf(“%d %d\n”,i,j);
printf(“%d,%d\n”,i,j);
printf(“i=%d,j=%d\n”,i,j);
printf(“%5d%5d\n”,i,j);
printf(“%-5d%-5d\n”,i,j);
float a=1.23;
printf(“%f\n”,a);
printf(“%5.3f\n”,a);
printf(“%5.1f\n”,a);
printf(“%-5.1f\n”,a);
float b=126.456789;
printf(“%e\n”,b);
printf(“%15.8e\n”,b);
printf(“%-15.1e\n”,b);
printf(“%%d\n”);
printf(“\\n”);
printf(“""\n”);
printf(“%f%%\n”,a);
return 0;
}
