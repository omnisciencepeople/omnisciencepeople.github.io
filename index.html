求一元二次方程解

#include<stdio.h>
#include<math.h>
int main()
{
float a,b,c,delta;
double x1,x2;
char ch;
do
{
printf(“请为一元二次方程的系数赋值(中间以空格相隔):”);
scanf(“%f %f %f”,&a,&b,&c);
delta=bb-4ac;
if(delta>0)
{
x1=(-b+sqrt(delta))/(2a);
x2=(-b-sqrt(delta))/(2a);
}
else if(delta==0)
{
x1=x2=(-b)/(2a);
}
else
printf(“无解\n”);
printf(“x1=%lf x2=%lf\n”,x1,x2);
printf(“请问你是否还要继续（Y/N）:”);
scanf(“%c”,&ch);
}while(ch==’Y’);
return 0;
}

{

{

#include<stdio.h>
int main()
{
int i;
float sum=0;
for(i=1;i<=100;i++)
sum=sum+1.0/i;//或者为sum=sum+1/float（i）；

如何看懂程序

分三步走：
第一步：流程控制，也就是程序的执行顺序；
第二步：语句功能，即这个语句种类及其作用；
第三步：试数，这个是看懂程序的关键；

在我们学习过程中可能会遇到一些小算法，我们可以尝试去对它编程，在解决不了的时候，可能会陷入自我怀疑中，这不是正确的学习方式；
这时候我们应该去看答案，重要的是把答案看懂，之后照着答案去敲程序并进行调试；如果可以，去修改程序，看看输出结果有什么不同；
如果连答案都看不懂，就把它背下来，不断去想，不断去敲，这是一个自我蜕变的过程，惟其坚持，方可成功！

对三个数字从小到大排序

#include<stdio.h>
int main()
{
int a,b,c;
printf(“请对这三个数赋值(中间以空格隔开):”);
scanf(“%d %d %d”,&a,&b,&c);
int t;
if(a>b)
{
t=a;
a=b;
b=t;
}
if(a>c)
{
t=a;
a=c;
c=t;
}
if(b>c)
{
t=b;
b=c;
c=t;
}
printf(“输出排序后的结果:%d %d %d\n”,a,b,c);
return 0;
}
