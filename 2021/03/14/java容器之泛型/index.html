java容器之泛型

我们编译程序时可能会出现“ 有关详细信息, 请使用 -Xlint:unchecked 重新编译。”这种情况，是因为我们没有限制一个容器或者接口中可以传入的数据类型；

```
import java.util.*;
public class test23{
public static void main(String[] args){
HashMap<student,Integer> hm=new HashMap<student,Integer> ();
hm.put(new student(“张三”,10,201208),1001);
hm.put(new student(“张三”,10,201208),1002);
hm.put(new student(“张三”,10,201208),1003);
hm.put(new student(“张三”,10,201208),1004);
Set S=hm.keySet();
Iterator it=S.iterator();
while(it.hasNext())
{
System.out.println(hm.get(it.next()));
}
System.out.println(hm.get(new student(“张三”,10,201208)));
System.out.println(hm.get(1001));
}
}
class student{
private int age;
private String name;
private int id;
public student(String name,int age,int id){
this.id=id;
this.age=age;
this.name=name;
}
public String toString(){
return name+” “+id+” “+age;
}
public int hashCode(){
return new String(name+id+age).hashCode();
}
public boolean equals(Object o){
student st=(student) o;
return st.id==this.id&&st.name==this.name&&st.age==this.age;
}
}
