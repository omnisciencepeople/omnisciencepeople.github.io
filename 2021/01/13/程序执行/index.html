#include<stdio.h>
int main()
{
int m,k=20;
m=(3>2)&&(k=8);
printf(“m=%d,k=%d\n”,m,k);
m=(1>2)&&(k=5);
printf(“m=%d,k=%d\n”,m,k);
m=(1>2)||(k=3);
printf(“m=%d,k=%d\n”,m,k);
m=(2>1)||(k=1);
printf(“m=%d,k=%d\n”,m,k);
m=(2>1)&&(k=0);
printf(“m=%d,k=%d\n”,m,k);
return 0;
}
