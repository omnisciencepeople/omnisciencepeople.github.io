#include<stdio.h>
int main()
{
int val;
int sum;
int m;
printf(“请输入你要判断的数字:”);
scanf(“%d”,&val);
m=val;
while(m)
{
sum=sum*10+m%10;
m/=10;
}
if(sum==val)
printf(“YES!\n”);
else
printf(“NO\n”);
return 0;
}
