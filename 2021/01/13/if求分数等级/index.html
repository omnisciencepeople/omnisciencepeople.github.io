if求分数等级

#include<stdio.h>
int main()
{
float score;
char ch;
do
{
printf(“请输入你的成绩:”);
scanf(“%f”,&score);
if(score>100)
printf(“做梦!\n”);
else if(80<=score&&score<=100)
printf(“优秀!\n”);
else if(70<=score&&score<80)
printf(“良好!\n”);
else if(60<=score&&score<70)
printf(“及格!\n”);
else if(0<=score&&score<60)
printf(“挂科!\n”);
else
printf(“你输入的成绩错误!\n”);
printf(“你是否还要继续（Y/N）:\n”);
scanf(“ %c”,&ch);//这里%c前加一个空格才可实现人机交互；
}while(ch==’Y’);
return 0;
}
