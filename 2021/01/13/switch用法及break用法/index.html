switch用法及break用法

#include<stdio.h>
int main()
{
int i;
printf(“请输入你想要进入的楼层:”);
scanf(“%d”,&i);
switch(i)
{
case 1:
printf(“1层开\n”);
break;
case 2:
printf(“2层开\n”);
break;
case 3:
printf(“3层开\n”);
break;
case 4:
printf(“4层开\n”);
break;
default:
printf(“输入楼层错误\n”);
break;
}
return 0;
}
/*
switch(i)
{
case 1:
A;
case 2:
B;
default:
C;
case 3:
D;
}
输入1，执行A B C D;
输入2，执行B C D;
输入3，执行D;
输入4，执行C D;
break 用法:
1.终止距离其最近的for循环；
2.终止switch语句；
3.终止if语句（此时if属于循环内部的语句）；
*/
