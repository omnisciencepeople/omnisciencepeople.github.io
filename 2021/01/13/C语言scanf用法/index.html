C语言scanf用法

#include <stdio.h>
#include <stdlib.h>
int main()
{
int i,j;
scanf(“%d,%d”&i,&j);
printf(“%d %d\n”,i,j);
scanf(“m%d %d\n”,&i,&j);
printf(“%d %d\n”,i,j);
scanf(“m%d %dn”,&i,&j);
printf(“%d %d\n”,i,j);
return 0;
}
