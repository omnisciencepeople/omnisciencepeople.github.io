求一个数是否是素数

#include<stdio.h>
int main()
{
int i,j;
printf(“请输入你要判断的数字:”);
scanf(“%d”,&i);
for(j=2;j<=i;j++)
if(i%j==0)
break;
if(i==j)
printf(“Yes!\n”);
else
printf(“No!”\n);
return 0;
}
