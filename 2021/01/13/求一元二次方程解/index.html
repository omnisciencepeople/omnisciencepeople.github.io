求一元二次方程解

#include<stdio.h>
#include<math.h>
int main()
{
float a,b,c,delta;
double x1,x2;
char ch;
do
{
printf(“请为一元二次方程的系数赋值(中间以空格相隔):”);
scanf(“%f %f %f”,&a,&b,&c);
delta=bb-4ac;
if(delta>0)
{
x1=(-b+sqrt(delta))/(2a);
x2=(-b-sqrt(delta))/(2a);
}
else if(delta==0)
{
x1=x2=(-b)/(2a);
}
else
printf(“无解\n”);
printf(“x1=%lf x2=%lf\n”,x1,x2);
printf(“请问你是否还要继续（Y/N）:”);
scanf(“%c”,&ch);
}while(ch==’Y’);
return 0;
}
