斐波拉契数列

#include<stdio.h>
int main()
{
int f1,f2,f3,n;
char ch;
f1=1;
f2=2;
do
{
printf(“请输入你要求的项数:”);
scanf(“%d”,&n);
if(n==1)
f3=f1;
else if(n==2)
f3=f2;
else
{
f3=f1+f2;
f1=f2;
f2=f3;
}
printf(“第%d项为%d\n”,n,f3);
printf(“你是否还要继续（Y/N）:”);
scanf(“ %c”,&ch);
}while(ch==’Y’);
return 0;
}
