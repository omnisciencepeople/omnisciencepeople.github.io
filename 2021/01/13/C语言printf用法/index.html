C语言printf用法

#include <stdio.h>
#include <stdlib.h>

int main()
{
printf(“Hello world!\n”);
int i=5,j=1;
printf(“%d\n”,i);
printf(“%d %d\n”,i,j);
printf(“%d,%d\n”,i,j);
printf(“i=%d,j=%d\n”,i,j);
printf(“%5d%5d\n”,i,j);
printf(“%-5d%-5d\n”,i,j);
float a=1.23;
printf(“%f\n”,a);
printf(“%5.3f\n”,a);
printf(“%5.1f\n”,a);
printf(“%-5.1f\n”,a);
float b=126.456789;
printf(“%e\n”,b);
printf(“%15.8e\n”,b);
printf(“%-15.1e\n”,b);
printf(“%%d\n”);
printf(“\\n”);
printf(“""\n”);
printf(“%f%%\n”,a);
return 0;
}
