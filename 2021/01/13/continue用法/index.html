continue用法

#include<stdio.h>
int main()
{
for(A;B;C)
{
D;
E;
continue;
F;
G;
}
return 0;
}
/*
执行顺序A->B->D->E->C->…
执行过程中跳过了F和G；
continue用于跳过本次循环余下的语句，转去判断是否执行下次循环；
*/
