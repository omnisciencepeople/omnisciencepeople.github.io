对三个数字从小到大排序

#include<stdio.h>
int main()
{
int a,b,c;
printf(“请对这三个数赋值(中间以空格隔开):”);
scanf(“%d %d %d”,&a,&b,&c);
int t;
if(a>b)
{
t=a;
a=b;
b=t;
}
if(a>c)
{
t=a;
a=c;
c=t;
}
if(b>c)
{
t=b;
b=c;
c=t;
}
printf(“输出排序后的结果:%d %d %d\n”,a,b,c);
return 0;
}
