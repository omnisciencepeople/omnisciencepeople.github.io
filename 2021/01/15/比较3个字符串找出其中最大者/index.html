比较3个字符串找出其中最大者

#include<stdio.h>
#include<string.h>
int main()
{
char str[3][20];//定义一个二维数组str，大小为3行20列，每一行可容纳一个字符串（最多为20个字符）；
char a[20];//定义一个一维数组作为中间变量交换字符串；
for(int i=0;i<3;i++)
gets(str[i]);//为每一行赋值
if(strcmp(str[0],str[1])>0)
strcpy(a,str[0]);
else
strcpy(a,str[1]);
if(strcmp(str[2],a)>0)
strcpy(a,str[2]);
printf(“%s\n”,a);
return 0;
}
