函数递归之阶乘

#include<stdio.h>
int f(int i)
{
if(i==1)
return 1;
else
return i*f(i-1);
}
int main()
{
int i;
char ch;
do
{
printf(“请输入你想求的阶乘数：”);
scanf(“%d”,&i);
printf(“%d的阶乘为%d\n”,i,f(i));
printf(“你是否还要继续（Y/N）:”);
scanf(“ %c”,&ch);
}while(ch==’Y’);
return 0;
}
