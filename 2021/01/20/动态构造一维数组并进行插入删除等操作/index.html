动态构造一维数组并进行插入删除等操作

#include<stdio.h>
#include<stdlib.h>
#include<malloc.h>
#include<stdbool.h>
typedef struct Arr
{
int* pBase;
int len;
int cnt;
}ARR,* PARR;
void init_arr(PARR,int);
void traverse_arr(PARR);
bool Is_empty(PARR);
bool Is_full(PARR);
bool append_arr(PARR,int);
bool Insert_arr(PARR,int,int);
bool Delete_arr(PARR,int,int*);
void sort_arr(PARR);
void Inversion_arr(PARR);
int main()
{
int val;
ARR arr;
init_arr(&arr,6);
traverse_arr(&arr);
append_arr(&arr,1);
append_arr(&arr,2);
append_arr(&arr,3);
append_arr(&arr,4);
traverse_arr(&arr);
Insert_arr(&arr,3,5);
Insert_arr(&arr,6,10);
traverse_arr(&arr);
if(Delete_arr(&arr,5,&val))
{
traverse_arr(&arr);
printf(“删除成功，你删除的元素为：%d\n”,val);
}
sort_arr(&arr);
traverse_arr(&arr);
Inversion_arr(&arr);
traverse_arr(&arr);
return 0;
}
void init_arr(PARR PArr,int len)
{
PArr->pBase=(int*)malloc(sizeof(int)len);
if(PArr->pBase==NULL)
{
printf(“动态构造数组失败程序终止\n”);
exit(-1);
}
else
{
PArr->len=len;
PArr->cnt=0;
}
return;
}
void traverse_arr(PARR PArr)
{
if(Is_empty(PArr))
printf(“数组为空”);
else
for(int i=0;icnt;i++)
printf(“%d “,PArr->pBase[i]);
printf(“\n”);
return;
}
bool Is_empty(PARR PArr)
{
if(PArr->cnt==0)
return true;
else
return false;
}
bool Is_full(PARR PArr)
{
if(PArr->cnt==PArr->len)
return true;
else
return false;
}
bool append_arr(PARR PArr,int val)
{
if(Is_full(PArr))
return false;
else
{
PArr->pBase[PArr->cnt]=val;
PArr->cnt++;
return true;
}
}
bool Insert_arr(PARR PArr,int pos,int val)//pos值从1开始
{
if(Is_full(PArr)||pos<1||pos>PArr->cnt+1)
return false;
else
{
for(int i=PArr->cnt-1;i>=pos-1;i–)
PArr->pBase[i+1]=PArr->pBase[i];
PArr->pBase[pos-1]=val;
PArr->cnt++;
return true;
}
}
bool Delete_arr(PARR PArr,int pos,int val)
{
if(Is_empty(PArr)||pos<1||pos>PArr->cnt)
return false;
else
{
*val=PArr->pBase[pos-1];
for(int i=pos-1;icnt-1;i++)
PArr->pBase[i]=PArr->pBase[i+1];
PArr->cnt–;
return true;
}
}
void sort_arr(PARR PArr)
{
int t;
for(int i=0;icnt-1;i++)
for(int j=0;jcnt-1-i;j++)
if(PArr->pBase[j]>PArr->pBase[j+1])
{
t=PArr->pBase[j];
PArr->pBase[j]=PArr->pBase[j+1];
PArr->pBase[j+1]=t;
}
return;
}
void Inversion_arr(PARR PArr)
{
int i=0;
int j=PArr->cnt-1;
int t;
while(i<j)
{
t=PArr->pBase[i];
PArr->pBase[i]=PArr->pBase[j];
PArr->pBase[j]=t;
i++;
j–;
}
return;
}
