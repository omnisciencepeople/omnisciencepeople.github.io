函数变量之静态局部变量

#include<stdio.h>
int f(int a);
int main()
{
int i,a=2;
for(i=1;i<=3;i++)
{
printf(“%d\n”,f(a));
}
return 0;
}
int f(int a)
{
int b=0;
static int c=3;
b++;
c++;
return (a+b+c);
}
输出：
7
8
9
