用布尔函数判断是否为素数

#include<stdio.h>
#include<stdbool.h>
bool Isprime(int val);
int main()
{
int val;
char ch;
do
{
printf(“请输入你要判断的数字:”);
scanf(“%d”,&val);
if(Isprime(val))
printf(“Yes\n”);
else
printf(“No\n”);
printf(“请问你是否还要继续（Y/N）:”);
scanf(“ %c”,&ch);
}while(ch==’Y’||ch==’y’);
return 0;
}
bool Isprime(int val)
{
for(int j=2;j<val;j++)
if(val%j==0)
break;
if(val==j)
return true;
else
return false;
}
