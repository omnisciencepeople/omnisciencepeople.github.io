二维数组行列互换

#include<stdio.h>
int main()
{
int a[2][3]={{1,2,3},{4,5,6}};
int b[3][2],i,j;
for(i=0;i<2;i++)
{
for(j=0;j<3;j++)
{
printf(“%d “,a[i][j]);
b[j][i]=a[i][j];
}
printf(“\n”);
}
printf(“互换后的结果:\n”);
for(i=0;i<3;i++)
{
for(j=0;j<2;j++)
printf(“%d “,b[i][j]);
printf(“\n”);
}
return 0;
}
