scanf输入字符串时注意事项

#include<stdio.h>
int main()
{
char a[100];
scanf(“%s”,&a);//error:字符数组中字符数组名即为该数组第一个元素的地址或者说该数组的起始地址；
scanf(“%s”,a)
printf(“%s\n”,a);//字符数组以%s输出时，输出控制项为字符数组名；
return 0;
}
#include<stdio.h>
int main()
{
char a[100],b[100],c[100];
scanf(“%s%s%s”,a,b,c);//若输入How are you?输出How are you?；若输入Howareyou?输出无输出;有空格字符分隔作为三个字符串输入，无空格字符，当作一个字符串输入；
两者a[100],b[100],c[100]不同
printf(“%s%s%s\n”,a,b,c);//字符数组以%s输出时，输出控制项为字符数组名；
char ch[100];
scanf(“%s”,ch);//若输入How are you？//输出How；系统将空格字符当作字符串间的分隔符故输出How
printf(“%s\n”,ch);
return 0;
}
//字符串的输出只需要字符数组首地址即可，系统会沿着第一个字符的地址找下去直至遇到\0；系统这么做的前提是数组为连续分配的内存；
疑问：
char ch[1或2];
scanf(“%s”,ch);//若输入How are you？为什么输出How；
printf(“%s\n”,ch);
