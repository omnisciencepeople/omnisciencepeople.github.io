continue用法

#include<stdio.h>
int main()
{
for(A;B;C)
{
D;
E;
continue;
F;
G;
}
return 0;
}
/*
执行顺序A->B->D->E->C->…
执行过程中跳过了F和G；
continue用于跳过本次循环余下的语句，转去判断是否执行下次循环；

switch用法及break用法

#include<stdio.h>
int main()
{
int i;
printf(“请输入你想要进入的楼层:”);
scanf(“%d”,&i);
switch(i)
{
case 1:
printf(“1层开\n”);
break;
case 2:
printf(“2层开\n”);
break;
case 3:
printf(“3层开\n”);
break;
case 4:
printf(“4层开\n”);
break;
default:
printf(“输入楼层错误\n”);
break;
}
return 0;
}
/*
switch(i)
{
case 1:
A;
case 2:
B;
default:
C;
case 3:
D;
}
输入1，执行A B C D;
输入2，执行B C D;
输入3，执行D;
输入4，执行C D;
break 用法:
1.终止距离其最近的for循环；
2.终止switch语句；
3.终止if语句（此时if属于循环内部的语句）；
*/

求一元二次方程解

#include<stdio.h>
#include<math.h>
int main()
{
float a,b,c,delta;
double x1,x2;
char ch;
do
{
printf(“请为一元二次方程的系数赋值(中间以空格相隔):”);
scanf(“%f %f %f”,&a,&b,&c);
delta=bb-4ac;
if(delta>0)
{
x1=(-b+sqrt(delta))/(2a);
x2=(-b-sqrt(delta))/(2a);
}
else if(delta==0)
{
x1=x2=(-b)/(2a);
}
else
printf(“无解\n”);
printf(“x1=%lf x2=%lf\n”,x1,x2);
printf(“请问你是否还要继续（Y/N）:”);
scanf(“%c”,&ch);
}while(ch==’Y’);
return 0;
}

如何看懂程序

分三步走：
第一步：流程控制，也就是程序的执行顺序；
第二步：语句功能，即这个语句种类及其作用；
第三步：试数，这个是看懂程序的关键；

在我们学习过程中可能会遇到一些小算法，我们可以尝试去对它编程，在解决不了的时候，可能会陷入自我怀疑中，这不是正确的学习方式；
这时候我们应该去看答案，重要的是把答案看懂，之后照着答案去敲程序并进行调试；如果可以，去修改程序，看看输出结果有什么不同；
如果连答案都看不懂，就把它背下来，不断去想，不断去敲，这是一个自我蜕变的过程，惟其坚持，方可成功！
