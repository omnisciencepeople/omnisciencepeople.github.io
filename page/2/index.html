一维数组冒泡排序

#include<stdio.h>
int main()
{
int i,j,t;
int a[10];
printf(“请为数组赋值(中间以空格相隔):\n”);
for(i=0;i<10;i++)
scanf(“%d”,&a[i]);
for(i=0;i<9;i++)
for(j=0;j<9-i;j++)
if(a[j]>a[j+1])
{
t=a[j];
a[j]=a[j+1];
a[j+1]=t;
}
printf(“排序后的结果为:\n”);
for(i=0;i<10;i++)
printf(“%d “,a[i]);
printf(“\n”);
return 0;

数组

/*
数组作用：为解决大量同类型数据的存储和使用；模拟现实世界；
数组分类:一维数组、二维数组、三维数组；
一维数组定义:为n个变量连续分配存储空间；所有变量数据类型相同；所有变量所占字节相同；
一维数组初始化:
定义时整体赋值，否则需单个赋值；
1.完全初始化
int a[3]={1,2,3};
输出1，2，3；
2.部分初始化:
int a[3]={1,2};
输出1，2 ，0；
未初始化值为0；
3.不初始化：
int a[3];
输出三个垃圾值；
一维数组的输出:
循环输出；
int a[n];//n表示元素的个数；元素为a[0]到a[n-1];
二维和三维数组大体同一维类似；
事实上，不存在多维数组，因为内存是线性一维的；
n维数组可以看作是每个元素是n-1维数组的一维数组；
例如:int a[3][4]可以看作为含有3个元素一维数组，只不过每个元素是含有4个元素的一维数组；
int a[3][4][5]可以看作是含有3个元素的一维数组，只不过每个元素是4行5列的二维数组；
*/

switch用法及break用法

#include<stdio.h>
int main()
{
int i;
printf(“请输入你想要进入的楼层:”);
scanf(“%d”,&i);
switch(i)
{
case 1:
printf(“1层开\n”);
break;
case 2:
printf(“2层开\n”);
break;
case 3:
printf(“3层开\n”);
break;
case 4:
printf(“4层开\n”);
break;
default:
printf(“输入楼层错误\n”);
break;
}
return 0;
}
/*
switch(i)
{
case 1:
A;
case 2:
B;
default:
C;
case 3:
D;
}
输入1，执行A B C D;
输入2，执行B C D;
输入3，执行D;
输入4，执行C D;
break 用法:
1.终止距离其最近的for循环；
2.终止switch语句；
3.终止if语句（此时if属于循环内部的语句）；

求一元二次方程解

#include<stdio.h>
#include<math.h>
int main()
{
float a,b,c,delta;
double x1,x2;
char ch;
do
{
printf(“请为一元二次方程的系数赋值(中间以空格相隔):”);
scanf(“%f %f %f”,&a,&b,&c);
delta=bb-4ac;
if(delta>0)
{
x1=(-b+sqrt(delta))/(2a);
x2=(-b-sqrt(delta))/(2a);
}
else if(delta==0)
{
x1=x2=(-b)/(2a);
}
else
printf(“无解\n”);
printf(“x1=%lf x2=%lf\n”,x1,x2);
printf(“请问你是否还要继续（Y/N）:”);
scanf(“%c”,&ch);
}while(ch==’Y’);
return 0;
}
